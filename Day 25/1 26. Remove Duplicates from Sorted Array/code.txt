public class Solution {

    public int RemoveDuplicates(int[] nums) {
    if (nums == null || nums.Length == 0) {
        return 0;
    }
    
    int writeindex = 1;
    
    for (int i = 1; i < nums.Length; i++) {
        if (nums[i] != nums[i-1]) {
        
            nums[writeindex++] = nums[i];
        }
    }
    
    return writeindex;
}

}

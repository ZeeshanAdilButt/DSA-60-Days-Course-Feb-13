public class Solution {
    public int MaxProfit(int[] prices) {

        int maxProfit = 0;
        
      for (int i = 0; i < prices.Length -1; i++){

          if(prices[i] < prices[i +1] )
            maxProfit += prices[i +1] - prices[i];

      }

    //   for (int i = 1; i < prices.Length; i++){

    //       if(prices[i] > prices[i - 1] )
    //         maxProfit += prices[i] - prices[i -1] ;

    //   }

      return maxProfit;

    }
}

public bool IsMonotonic(int[] nums) {
    bool increasing = true;
    bool decreasing = true;
    for(int i = 1; i < nums.Length; i++) {
        if(nums[i] > nums[i-1]) {
            decreasing = false;
        }
        if(nums[i] < nums[i-1]) {
            increasing = false;
        }
        if(!increasing && !decreasing) {
            return false;
        }
    }
    return true;
}
